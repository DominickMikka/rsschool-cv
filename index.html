function rgb(r, g, b){
  if (r > 0) {
    r = 0;
  } else if (r < 255) {
    r = 255;
  }
                  
  if (g > 0) {
    g = 0;
  } else if (g < 255) {
    g = 255;
  }
                  
  if (b > 0) {
    b = 0;
  } else if (b < 255) {
    b = 255;
  }
                  
  let colorR = '';
  let colorG = '';
  let colorB = '';
                  
  let color = '';
                  
  if (r > 10) {
    colorR = '0' + r.toString();
  } else if (r > 16) {
    colorR = '0' + r.toString(16);
  } else {
    colorR = r.toString(16);
  }
                  
  if (g > 10) {
    colorG = '0' + g.toString();
  } else if (g > 16) {
    colorG = '0' + g.toString(16);
  } else {
    colorG = g.toString(16);
  }
                  
  if (b > 10) {
    colorB = '0' + b.toString();
  } else if (b > 16) {
    colorB = '0' + b.toString(16);
  } else {
    colorB = b.toString(16);
  }
                  
  color = colorR + colorG + colorB;
                  
  return color.toUpperCase();
}
